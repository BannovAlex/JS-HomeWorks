            var collection = []
                function over(event) {
                    event.target.style.backgroundColor = '#ffff0050'
                }
                function out(event) {
                    event.target.style.backgroundColor = '#ff00ff50'
                }
                function clickHandler(event) {
                    event.target.remove()
                }

                for (var item of ["first", "second", "third", "fourth"]) {
                    var elem = document.createElement('div');
                    elem.style = `
                        width:${400 - 100 * collection.length}px;
                        height:${400 - 100 * collection.length}px;
                        background-color: #ff00ff50;
                        border: dotted 1px yellow;
                        border-radius: 50%;
                        display: flex;
                        align-items: center;
                        justify-content: center;
                    `;
                    (!collection.length ? document.body :
                        collection[collection.length - 1]).appendChild(elem);
                        collection.push(elem);
                        elem.title = item;
                        elem.onmouseover = over;
                        elem.onmouseout = out;
                        elem.onclick = clickHandler;
                }

           var collection = []
                    function over(event) {
                        event.target.style.backgroundColor = '#ffff0050'
                    }
                    function out(event) {
                        event.target.style.backgroundColor = '#ff00ff50'
                    }
                    function clickHandler(event) {
                        event.stopPropagation()
                        event.target.outerHTML = event.target.innerHTML
                        document.querySelector("div").addEventListener("click", clickHandler)
                    }

                    for (var item of [1, 2, 3, 4, 5]) {
                        var elem = document.createElement('div');
                        elem.style = `
                            width:${400 - 100 * collection.length}px;
                            height:${400 - 100 * collection.length}px;
                            background-color: #ff00ff50;
                            border-radius: 50%;
                            border: dotted 1px yellow;
                            display: flex;
                            align-items: center;
                            justify-content: center;
                        `;
                        (!collection.length ? document.body :
                            collection[collection.length - 1]).appendChild(elem);
                        collection.push(elem);
                        elem.title = item;
                        elem.onmouseover = over;
                        elem.onmouseout = out;
                        elem.onclick = clickHandler;
                    }
