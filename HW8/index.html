        var p = document.body.appendChild(
            document.createElement("p")
        )
        var interval = 100
        function currentTime() {
            interval--
            if ( interval < 0 ) return p.remove()
            p.innerText = new Date().toLocaleTimeString()
            setTimeout ( currentTime, 1000 )
        }
        currentTime()
